    { id: 'delivery', label: '美国站配送费计算', icon: Truck },

  const renderContent = () => {
    if (activeTab === 'home') return <HomePage onNavigate={setActiveTab} />;
    if (activeTab === 'ad-calc') return <AdCalculatorPage />;
    if (activeTab === 'unit') return <UnitConverterPage />;
    const currentItem = menuItems.find(item => item.id === activeTab);
    return (
      <PlaceholderPage 
        title={currentItem?.label || '未知模块'}

    );
  };

  return (
    <div className="min-h-screen bg-slate-50 flex flex-col">

      <header className="h-14 bg-[#5b5bd6] text-white flex items-center px-6 shadow-md z-20">

        <aside className="w-64 bg-white border-r border-gray-200 flex-shrink-0 hidden md:flex flex-col">
          <div className="p-4 space-y-1 flex-1 overflow-y-auto">
            {menuItems.map((item) => (
              <button
                key={item.id}
                onClick={() => setActiveTab(item.id)}

                    ? 'bg-blue-50 text-blue-600' 
                    : 'text-gray-600 hover:bg-gray-50 hover:text-gray-900'

              >
